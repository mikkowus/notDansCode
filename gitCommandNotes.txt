###git clone https://github.com/user/repo.git
git clone https://github.com/mikkowus/notDansCode.git
git clone --branch branch1 https://github.com/mikkowus/notDansCode.git
git clone --branch branch1 --single-branch https://github.com/mikkowus/notDansCode.git
git branch branch1

##creates new branch called branch1
git checkout -b newBranchName

##Shows you what branch you are in
git branch 

##switches you over to branch1
git checkout branch1


##these 5 commands in sequence will upload your code to the remote git repo
##Be sure the active directory (when executing these commands) contains the files to be uploaded. 
git config --global user.email "you@example.com"
git config --global user.name "Your Name"
git add .
git commit -m "a commit message"
git push origin branch1

##pulls all new changes down from the remote git repo
git pull
